pixelToBitmap.js

High performance converter of pixel array to bitmap image.

Example

6x6 pixel array to Bitmap Image.

#ff0000#ff0000#ff0000#00ff00#00ff00#00ff00
#ff0000#ff0000#ff0000#00ff00#00ff00#00ff00
#ff0000#ff0000#ff0000#00ff00#00ff00#00ff00
#0000ff#0000ff#0000ff#ff00ff#ff00ff#ff00ff
#0000ff#0000ff#0000ff#ff00ff#ff00ff#ff00ff
#0000ff#0000ff#0000ff#ff00ff#ff00ff#ff00ff
Convert

How to Use

var colorArr = [
  ['#ff0000', '#ff0000', '#ff0000', '#00ff00', '#00ff00', '#00ff00'],
  ['#ff0000', '#ff0000', '#ff0000', '#00ff00', '#00ff00', '#00ff00'],
  ['#ff0000', '#ff0000', '#ff0000', '#00ff00', '#00ff00', '#00ff00'],
  ['#0000ff', '#0000ff', '#0000ff', '#ff00ff', '#ff00ff', '#ff00ff'],
  ['#0000ff', '#0000ff', '#0000ff', '#ff00ff', '#ff00ff', '#ff00ff'],
  ['#0000ff', '#0000ff', '#0000ff', '#ff00ff', '#ff00ff', '#ff00ff']
];

var image = pixelToBitmap(colorArr);
var div = document.getElementById('converted_result');
div.appendChild(image);
